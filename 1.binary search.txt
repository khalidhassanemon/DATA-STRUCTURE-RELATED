#include<bits/stdc++.h>
using namespace std;
int main()
{   int n,x,i;
    cin>>n;
    int A[n+1];
    for(i=0; i<n; i++)
    {
        cin>>A[i];
    }
    cin>>x;
    int left=0;
    int right=n-1;
    while(left<=right)
    {
        int mid=(left+right)/2;
        if(A[mid]<x)
        {
            left=mid+1;
        }
        else
            right=mid-1;

    }
    if(A[mid]==x)
        {
            cout<<mid;
        }
  return -1;
}
