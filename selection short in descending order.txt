#include<bits/stdc++.h>
using namespace std;
int main()
{
    int i,j,n,temp=0;
    cin>>n;
    int a[n+10];
    for(i=0; i<n; i++)
        cin>>a[i];
    for(j=0; j<n-1; j++)
    {
        for(i=j+1; i<n; i++)
        {
            if(a[j]<a[i])
            {
                temp=a[j];
                a[j]=a[i];
                a[i]=temp;
            }
        }
    }
    for(i=0; i<n; i++)
        cout<<a[i]<<" ";
}
